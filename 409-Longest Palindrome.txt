public class Solution {
    public int longestPalindrome(String s) {
        int[] arr = new int[58];
		for(int i=0;i<s.length();i++){
			arr[s.charAt(i)-'A']++;
		}
		int cnt = 0;
		int cntodd = 0;
		for(int i=0;i<58;i++){
			if(arr[i]%2!=0)
				cntodd++;
			cnt += arr[i];
		}
		if(cntodd <= 1)
			return cnt;
		return cnt-cntodd+1;
    }
}
